1. 启动tomcat

docker run -d --name mytomcat -p 80:8080 -v /root/webapps:/usr/local/tomcat/webapps --restart always tomcat:8.5

2. 启动mysql容器或是mariadb容器

docker run -d --name mysql -p 3306:3306 -e MYSQL_ROOT_PASSWORD=root mysql:5.7

docker run -d --name mariadb -p 3306:3306 -e MYSQL_ROOT_PASSWORD=root mariadb:10.2

3. 启动redis容器

docker run -d --name redis -p 6379:6379 redis:5.0

docker run -d --name redis -p 6379:6379 redis:5.0 redis-server --appendonly yes

4. 启动gitlab容器

sudo docker run --detach \
  --hostname gitlab.example.com \
  --publish 443:443 --publish 80:80 --publish 22:22 \
  --name gitlab \
  --restart always \
  --volume  $HOME/gitlab/config:/etc/gitlab \
  --volume  $HOME/gitlab/logs:/var/log/gitlab \
  --volume  $HOME/gitlab/data:/var/opt/gitlab \
  gitlab/gitlab-ce:latest

5. 启动docker容器管理图形化界面

docker run -d -p 9000:9000 --restart=always -v /var/run/docker.sock:/var/run/docker.sock --name portainer 62771b0b9b09

docker run -it -d --name dockerui -p 9000:9000 -v /var/run/docker.sock:/var/run/docker.sock docker.io/uifd/ui-for-docker

6. 启动nginx容器

docker run -d --name nginx --restart=always -p 80:80 --privileged=true  
-v /cloud/nginx/webroot:/usr/share/nginx/html 
-v /cloud/nginx/config/nginx.conf:/etc/nginx/nginx.conf:ro 
-v /cloud/nginx/config/conf.d:/etc/nginx/conf.d 
-v /cloud/nginx/log:/var/log/nginx 
nginx

7. 启动mongo容器

docker run --name mongo -p 27017:27017 -v ~/db:/data/db -d mongo:latest

8. 启动nacos容器

docker run -d --name mynacos -p 8858:8848 -e MODE=standalone  docker.io/tanyi/nacos-server

9. 启动consul容器

docker run -d --name consul -p 8500:8500 consul

10. 启动zookeeper容器

docker run --name zookeeper3.4.9 -p 2181:2181 --restart always -d zookeeper:3.4.9

启动zk的客户端
docker exec -it 容器id zkCli.sh
